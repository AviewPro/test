<script>
    let aiCandidates = [], history = [], inputStr = "", turnCount = 7;
    const bgm = new Audio("https://cdn.discordapp.com/attachments/1471931935969247347/1472065317822791813/002._ErikMMusic_-_The_Dusty_Attic.mp3?ex=69913729&is=698fe5a9&hm=e2772ad864c0d14a86305999c627b537204bf7a6d971e90b7fe6f5f1a8588cc0&");
    bgm.loop = true;
    bgm.playbackRate = 2.0;
    const clickSfx = new Audio("https://assets.mixkit.co/active_storage/sfx/2568/2568-preview.mp3");
    const meowSfx = new Audio("https://cdn.discordapp.com/attachments/1471931935969247347/1471932047802105927/moeeza3-kitten-cat-meow-437244.mp3?ex=699163cb&is=6990124b&hm=8253c2ec2e225d147fa9358b0d7319f23762439eef3e010b1efb77091bdb9ad2&");

    function adjustVolume(val) { 
        bgm.volume = val; 
        meowSfx.volume = val; 
        localStorage.setItem('quantum_volume', val); 
    }
    function playClick() { const sc = clickSfx.cloneNode(); sc.play().catch(e=>{}); if (bgm.paused) bgm.play().catch(e=>{}); }

    function startGame() {
        playClick();
        document.getElementById('setup').style.display = 'none';
        document.getElementById('game-box').style.display = 'flex';
        document.getElementById('info-panel').style.display = 'flex';
        aiCandidates = [];
        for(let i=0; i<=9999; i++){
            let n = i.toString().padStart(4, '0');
            if(new Set(n).size === 4) aiCandidates.push(n);
        }
        updateUI();
    }

    function aiRespond(guess) {
        const outcomes = {};
        aiCandidates.forEach(cand => {
            let s=0, b=0;
            for(let i=0; i<4; i++){ if(guess[i] === cand[i]) s++; else if(cand.includes(guess[i])) b++; }
            const key = `${s}S${b}B`;
            if(!outcomes[key]) outcomes[key] = [];
            outcomes[key].push(cand);
        });
        let bestKey = "";
        let maxCount = -1;
        for(let key in outcomes) {
            if(key === "4S0B") continue; 
            if(outcomes[key].length > maxCount) { maxCount = outcomes[key].length; bestKey = key; }
        }
        if(!bestKey || aiCandidates.length <= 1) {
            const fakeS = Math.floor(Math.random() * 3);
            const fakeB = Math.floor(Math.random() * 3);
            bestKey = `${fakeS}S${fakeB}B`;
            aiCandidates = aiCandidates.filter(c => c !== guess); 
        } else { aiCandidates = outcomes[bestKey]; }
        const match = bestKey.match(/(\d)S(\d)B/);
        return { s: parseInt(match[1]), b: parseInt(match[2]) };
    }

    function updateUI() {
        const turnCounter = document.getElementById('turn-counter');
        turnCounter.innerText = turnCount;
        const prob = aiCandidates.length === 0 ? 0 : (1 / aiCandidates.length * 100);
        const chanceValEl = document.getElementById('chance-val');
        const chanceTitleEl = document.getElementById('chance-title');
        const chanceBoxEl = document.getElementById('chance-box');
        const catImg = document.getElementById('prob-cat-img');
        
        if(aiCandidates.length === 0) {
            chanceTitleEl.style.display = 'none';
            chanceValEl.innerHTML = `<span class="poison-text" style="color:var(--green); text-shadow: 0 0 15px var(--green);">Cat is Poisoned<br>in the Box</span>`; 
            chanceBoxEl.style.borderColor = 'var(--green)';
            catImg.style.opacity = '0';
            catImg.style.filter = 'blur(20px)';
            turnCounter.style.color = 'var(--green)';
            turnCounter.style.textShadow = '0 0 20px var(--green)';
        } else {
            chanceTitleEl.style.display = 'block';
            chanceTitleEl.innerText = "Measuring Energy Level";
            chanceValEl.innerText = prob.toFixed(2) + "%";
            chanceTitleEl.style.color = 'var(--purple)';
            chanceValEl.style.color = 'var(--purple)';
            chanceBoxEl.style.borderColor = 'var(--purple)';
            turnCounter.style.color = 'var(--purple)';
            turnCounter.style.textShadow = '0 0 20px var(--purple)';
            const blurAmount = Math.max(0, (prob * 0.5));
            const opacityAmount = Math.max(0.1, 1 - (prob / 100));
            catImg.style.display = 'block';
            catImg.style.opacity = opacityAmount;
            catImg.style.filter = `blur(${blurAmount}px)`;
            catImg.style.borderColor = 'var(--purple)';
            catImg.style.boxShadow = 'none';
        }
        
        document.getElementById('log-window').innerHTML = history.map(h => {
            const resultText = (h.s === 0 && h.b === 0) ? `<span class="log-out">OUT</span>` : `${h.s}S / ${h.b}B`;
            return `<div class="log-item"><span class="log-n">${h.n}</span><span class="log-sb">${resultText}</span></div>`;
        }).join('');
        document.getElementById('log-window').scrollTop = document.getElementById('log-window').scrollHeight;

        const slice = [...aiCandidates].sort(() => 0.5 - Math.random()).slice(0, 50);
        const pixelateStrength = prob >= 25 ? 0 : (25 - prob) / 3.5; 
        document.getElementById('cand-list').innerHTML = slice.map(c => `<div class="cand-item" style="filter: blur(${pixelateStrength}px) contrast(180%);">${c}</div>`).join('');

        const entBtn = document.getElementById('ent-btn');
        if(turnCount <= 0) {
            entBtn.disabled = true;
            document.getElementById('status').innerText = "측정 소진 (관측만 가능)";
        } else {
            entBtn.disabled = false;
            document.getElementById('status').innerText = "고양이 위치 관측 중...";
        }
    }

    function press(n) { playClick(); if(inputStr.length < 4 && !inputStr.includes(n)) { inputStr += n; document.getElementById('display').innerText = inputStr; } }
    function clearIn() { playClick(); inputStr = ""; document.getElementById('display').innerText = "----"; }
    
    function clickEnter() {
        if(inputStr.length === 4 && turnCount > 0) {
            playClick();
            const res = aiRespond(inputStr);
            history.push({ n: inputStr, s: res.s, b: res.b });
            turnCount--;
            clearIn();
            updateUI();
        }
    }

    function renderProbabilityCloud() {
        const field = document.getElementById('quantum-field');
        field.innerHTML = "";
        const neons = ['var(--cyan)', 'var(--purple)', 'var(--green)', 'var(--red)'];
        const list = aiCandidates.length > 0 ? aiCandidates : ["ERROR", "VOID", "0000"];
        list.slice(0, 80).forEach(num => {
            const span = document.createElement('span');
            span.className = 'bg-candidate';
            span.innerText = num;
            span.style.left = (Math.random() * 90 + 5) + "%";
            span.style.top = (Math.random() * 90 + 5) + "%";
            span.style.color = neons[Math.floor(Math.random() * neons.length)];
            field.appendChild(span);
